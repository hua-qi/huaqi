```typescript
export function useEventLog(urls: string[]) {
  const [entries, setEntries] = useState<EventLogEntry[]>([]);

  useEffect(() => {
    const sources = urls.map((url) => {
      const es = new EventSource(url);
      es.onmessage = (e) => {
        try {
          const entry: EventLogEntry = JSON.parse(e.data);
          setEntries((prev) => [...prev.slice(-99), entry]);
        } catch {}
      };
      return es;
    });
    return () => sources.forEach((es) => es.close());
  }, []);

  return entries;
}
```

## 六、关键发现与结论

### 结论 1: **两种获取方式都已实现**

| 获取方式 | 状态 | 实现位置 |
|---------|------|--------|
| **HTTP Endpoint** | ✅ **完整实现** | CF-Agent `/resource-proxy` → SG-Agent `/mcp/resources/read` |
| **MCP Server SSE/JSON-RPC** | ⚠️ **部分实现** | stargate-mcp-ui-server 通过 FastMCP 暴露工具和资源，但前端未直接消费 |

---

### 结论 2: **A2A 响应中的 resourceUri 流转链路**

1. **生成**: SG-Agent 在工具调用时生成 `ui://stargate/card/{cardId}` 格式的 URI
2. **嵌入**: 在 A2A JSON 响应中通过 `mcp_ui_resource.resourceUri` 字段包含
3. **透传**: CF-Agent 完全透传，无任何解析或转换
4. **消费**: 前端通过 `CardMessage` 组件的 `onReadResource` 回调获取内容

---

### 结论 3: **前端获取 Resource 内容的两条路径**

**路径 A (现状 - HTTP Endpoint)**:
```
前端 → CF-Agent /resource-proxy → SG-Agent /mcp/resources/read → HTML
```

**路径 B (备选 - MCP SSE)**:
```
前端 → MCP SSE 订阅 stargate-mcp-ui-server → resource 内容 → HTML
```

当前前端使用**路径 A**，因为：
- CF-Agent 已有代理端点
- 前端 AppRenderer 通过 `onReadResource` 回调实现
- 无需直接连接 MCP Server

---

### 结论 4: **核心架构优势**

```
┌─────────────────────────────────┐
│    A2A 协议 (Agent-Agent)       │  ← ResourceUri 仅作为指针，无实际数据
├─────────────────────────────────┤
│    CF-Agent 透传层              │  ← 不解析资源内容，只做消息路由
├─────────────────────────────────┤
│    HTTP Endpoint (代理)          │  ← 按需获取资源内容（分离关注点）
├─────────────────────────────────┤
│    SG-Agent Resource Manager    │  ← 管理缓存、鉴权、数据脱敏
├─────────────────────────────────┤
│    前端 CardMessage             │  ← iframe 沙箱隔离 + 组件自治
└─────────────────────────────────┘
```
