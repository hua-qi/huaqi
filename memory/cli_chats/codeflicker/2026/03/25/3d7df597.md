## React 使用方式

原生支持，提供 `renderAsync` API：

```tsx
import { renderAsync } from 'docx-preview';
import { useRef, useEffect } from 'react';

function DocxViewer({ file }) {
  const containerRef = useRef(null);

  useEffect(() => {
    if (file && containerRef.current) {
      renderAsync(file, containerRef.current);
    }
  }, [file]);

  return <div ref={containerRef} />;
}
```

## 特点

| 方面 | 评价 |
|------|------|
| 框架绑定 | 无，原生 JS 库 |
| React 适配 | 简单，仅需 `useRef` |
| 社区封装 | 有 `react-docx-preview` 等 wrapper |

## 注意事项

- 需要处理 `Buffer/ArrayBuffer` 转换
- 样式通过注入 `<style>` 实现，注意 CSS 隔离

[2026-03-25 12:02:43] [assistant]: ```tsx
import { useState, useRef } from 'react';
import { renderAsync } from 'docx-preview';

function DocxViewer() {
  const [loading, setLoading] = useState(false);
  const containerRef = useRef<HTMLDivElement>(null);

  const handleFileChange = async (e: React.ChangeEvent<HTMLInputElement>) => {
    const file = e.target.files?.[0];
    if (!file || !containerRef.current) return;

    setLoading(true);
    try {
      const arrayBuffer = await file.arrayBuffer();
      
      // 清空之前的内容
      containerRef.current.innerHTML = '';
      
      await renderAsync(arrayBuffer, containerRef.current, undefined, {
        className: 'docx-preview', // 自定义 class
        inWrapper: false,           // 是否包裹在 div 中
        ignoreWidth: false,         // 忽略宽度
        ignoreHeight: false,        // 忽略高度
        ignoreFonts: false,         // 忽略字体
        breakPages: true,           // 分页显示
        ignoreLastRenderedPageBreak: true,
        trimXmlDeclaration: true,
        useBase64URL: false,
        renderChanges: false,       // 显示修订痕迹
        renderComments: false,      // 显示批注
      });
    } catch (error) {
      console.error('渲染失败:', error);
    } finally {
      setLoading(false);
    }
  };

  return (
    <div>
      <input 
        type="file" 
        accept=".docx" 
        onChange={handleFileChange} 
      />
      {loading && <p>加载中...</p>}
      <div 
        ref={containerRef} 
        style={{ 
          marginTop: 20,
          border: '1px solid #ddd',
          padding: 20,
          minHeight: 400,
          overflow: 'auto'
        }} 
      />
    </div>
  );
}

export default DocxViewer;
```

**关键配置项：**
- `breakPages: true` - 保留分页效果
- `inWrapper: false` - 更灵活的控制样式
- `renderChanges/renderComments` - 按需显示修订和批注

**样式隔离建议：**
```css
.docx-preview {
  font-family: 'SimSun', serif;
}
.docx-preview p {
  margin: 0;
}
