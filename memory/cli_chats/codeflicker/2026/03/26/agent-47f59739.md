#### 1.1 MarkdownChunksCard 分段渲染机制

**核心实现代码：**

```typescript
const MarkdownChunksCard = ({
  cardContent,
  isGenerating
}: {
  cardContent: MarkdownCardMetaData;
  isGenerating: boolean;
}) => {
  // 分段渲染Markdown
  const [markdownChunks, setMarkdownChunks] = useState<string[]>([]);

  useEffect(() => {
    const content = cardContent?.content || '';
    if (!content) return;

    function splitMarkdownSafe(input: string): string[] {
      const parts: string[] = [];
      let buffer = '';
      let insideCodeBlock = false;

      const lines = input.split('\n');
      for (let i = 0; i < lines.length; i++) {
        const line = lines[i];

        // 根据 ``` 拆成语义块
        if (line.trim().startsWith('```')) {
          if (insideCodeBlock) {
            buffer += (buffer ? '\n' : '') + line;
            parts.push(buffer);
            buffer = '';
            insideCodeBlock = false;
          } else {
            if (buffer.trim()) {
              parts.push(buffer);
            }
            buffer = line;
            insideCodeBlock = true;
          }
        } else if (insideCodeBlock) {
          buffer += '\n' + line;
        } else {
          // 根据 \n\n 拆成语义块
          if (line.trim() === '' && buffer !== '') {
            parts.push(buffer);
            buffer = '';
          } else {
            buffer += (buffer ? '\n' : '') + line;
          }
        }
      }

      if (buffer.trim()) {
        parts.push(buffer);
      }

      return parts;
    }

    // 直接对完整 content 做一次安全切分
    const newChunks = splitMarkdownSafe(content);
    setMarkdownChunks(newChunks);
  }, [cardContent?.content]);

  return (
    <div className="markdown-card-list-container">
      {markdownChunks?.map((c, index) => (
        <CachedMarkdownCard
          key={`${cardContent?.index}+${index}`}
          content={c || ''}
          index={`${cardContent?.index}+${index}`}
          disableCitations={isGenerating}
        />
      ))}
    </div>
  );
};

export default MarkdownChunksCard;
```

**分段渲染的三个关键特点：**

1. **智能分割算法：** `splitMarkdownSafe()` 函数会根据以下规则分割 Markdown：
   - 检测代码块边界（```)
   - 检测段落边界（双换行符 `\n\n`）
   - 保证代码块内的内容完整性

2. **分段渲染的好处：**
   - 避免一次性渲染整个 Markdown，减轻页面渲染压力
   - 每个分段都通过 `CachedMarkdownCard` 独立渲染
   - 支持流式更新时的增量显示

3. **关键配置：**
   - `disableCitations={isGenerating}` - 在生成过程中禁用引用功能，防止中间态问题

// 改进版 splitMarkdownSafe 函数
function splitMarkdownSafeWithLinkCheck(input: string): string[] {
  const parts: string[] = [];
  let buffer = '';
  let insideCodeBlock = false;

  const lines = input.split('\n');
  for (let i = 0; i < lines.length; i++) {
    const line = lines[i];

    if (line.trim().startsWith('```')) {
      if (insideCodeBlock) {
        buffer += (buffer ? '\n' : '') + line;
        parts.push(buffer);
        buffer = '';
        insideCodeBlock = false;
      } else {
        if (buffer.trim()) {
          parts.push(buffer);
        }
        buffer = line;
        insideCodeBlock = true;
      }
    } else if (insideCodeBlock) {
      buffer += '\n' + line;
    } else {
      if (line.trim() === '' && buffer !== '') {
        // 检查 buffer 末尾是否有不完整的链接
        if (!hasIncompleteLink(buffer)) {
          parts.push(buffer);
          buffer = '';
        } else {
          // 延迟分割，等待下一行补完
          buffer += '\n' + line;
        }
      } else {
        buffer += (buffer ? '\n' : '') + line;
      }
    }
  }

  if (buffer.trim()) {
    parts.push(buffer);
  }

  return parts;
}

// 检查是否有不完整的链接
function hasIncompleteLink(text: string): boolean {
  // 检测模式：[...]( 后面没有 )
  const linkPattern = /\[[^\]]*\]\([^)]*$/;
  return linkPattern.test(text);
}
