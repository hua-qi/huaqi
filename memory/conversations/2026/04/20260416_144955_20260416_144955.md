### **步骤2：创建第一个示例文件（8分钟）**
创建 `day1_hello_langgraph.py`：

```python
#!/usr/bin/env python3
"""
Day 1: LangGraph 第一个示例
学习目标：了解最基本的 Graph 使用
"""

from langgraph.graph import StateGraph, END
from typing import TypedDict

# 1. 定义状态类型
class MyState(TypedDict):
    message: str
    counter: int

# 2. 创建图构建器
builder = StateGraph(MyState)

# 3. 定义节点函数
def node_a(state: MyState) -> MyState:
    """第一个节点：初始化消息"""
    print("执行节点 A")
    return {"message": "Hello LangGraph!", "counter": state.get("counter", 0) + 1}

def node_b(state: MyState) -> MyState:
    """第二个节点：处理消息"""
    print("执行节点 B")
    message = state["message"]
    counter = state["counter"]
    return {"message": f"{message} (已处理 {counter} 次)", "counter": counter}

# 4. 添加节点到图中
builder.add_node("node_a", node_a)
builder.add_node("node_b", node_b)

# 5. 设置入口点
builder.set_entry_point("node_a")

# 6. 添加边（定义执行流程）
builder.add_edge("node_a", "node_b")
builder.add_edge("node_b", END)

# 7. 编译图
graph = builder.compile()

# 8. 执行图
print("\n=== 开始执行 ===")
initial_state = {"message": "", "counter": 0}
result = graph.invoke(initial_state)

print("\n=== 执行结果 ===")
print(f"最终消息: {result['message']}")
print(f"执行次数: {result['counter']}")

# 9. 可视化图（可选）
try:
    # 需要安装 graphviz
    from langgraph.graph import get_graph
    display = get_graph(graph)
    print("\n图结构已生成（如果安装了graphviz）")
except:
    print("\n（跳过可视化，需要安装graphviz）")
```
